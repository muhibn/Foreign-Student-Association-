create database FSA
  create table  Admin (
    AUID INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	ANAME VARCHAR(50) NOT NULL UNIQUE,
	APASSWORD VARCHAR(15) NOT NULL UNIQUE );
  
  
  create table userpersonal(
    SUID INT(30) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	SNAME VARCHAR(50) NOT NULL,
	LNAME VARCHAR(50) NOT NULL,
	GENDER VARCHAR(10) NOT NULL,
	EMAIL VARCHAR(40) NOT NULL UNIQUE,
	UNAME VARCHAR(50) NOT NULL  UNIQUE,
	SPASSWORD VARCHAR(50) NOT NULL UNIQUE,
	AUID INT(11) DEFAULT NULL,
	FOREIGN KEY(AUID) REFERENCES  admin(AUID)
	);
  create table usercourse(
    CUID INT(30) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	COURSE VARCHAR(50),
	COLLEGE VARCHAR(50),
	USNIVERSITY VARCHAR(50),
	DATABIRTH DATE NOT NULL,
	SUID INT(30) NOT NULL,
AUID    INT(11) DEFAULT NULL,
	FOREIGN KEY(AUID) REFERENCES  admin(AUID),
	FOREIGN KEY(SUID) REFERENCES USNIVERSITY(SUID)
	);
  create table post(
    PUID INT(30) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    IMAGE BLOB,
	TEXT LONGTEXT,
	PDATA DATETIME DEFAULT NOW(),
	SECTION VARCHAR(40) NOT NULL,
	SUID INT(30),
	AUID INT(11),
	FOREIGN KEY(SUID) REFERENCES userpersonal(SUID),
	FOREIGN KEY(AUID) REFERENCES ADMIN(AUID)
	);
  create table DOCUMENT(
    DUID INT(30) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	PDATA DATETIME DEFAULT NOW(),
	SECTION VARCHAR(40) NOT NULL,
	DNAME VARCHAR(50) NOT NULL,
	SUID INT(30),
	AUID INT(11),
	FOREIGN KEY(SUID) REFERENCES userpersonal(SUID),
	FOREIGN KEY(AUID) REFERENCES ADMIN(AUID)
	);
  create table comment(
    COMUID INT(30) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	COMMENT TINYTEXT ,
	PUID INT(30),
	SUID INT(30),
	AUID INT(11),
	FOREIGN KEY(PUID) REFERENCES POST(PUID),
	FOREIGN KEY(SUID) REFERENCES userpersonal(SUID),
	FOREIGN KEY(AUID) REFERENCES ADMIN(SUID)
	);
